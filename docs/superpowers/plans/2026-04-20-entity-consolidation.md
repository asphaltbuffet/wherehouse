# Entity Consolidation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the separate `item` and `location` types with a unified `entity` model backed by a single `entities_current` projection table and an `entity.*` event namespace.

**Architecture:** A new `EntityType` iota enum (`place`, `container`, `leaf`) and `EntityStatus` iota enum (`ok`, `borrowed`, `missing`, `loaned`, `removed`) replace the item/location split. All events move to the `entity.` prefix. A single `entities_current` table stores all entities with parent-child hierarchy, full path computation, and first-class status. CLI commands are unified — `add`, `move`, `status`, `rename`, `remove`, `scry`, `list`, `history` — operating on entity IDs (nanoids).

**Tech Stack:** Go, SQLite (via `database/sql` + `modernc.org/sqlite`), `cobra` CLI, `golang-migrate` migrations, `nanoid` (internal package), `lipgloss` styles, `testify`, `mockery`.

**Spec:** `docs/superpowers/specs/2026-04-20-entity-consolidation-design.md`

```go
	EntityTypeContainer // container

	EntityTypeLeaf // leaf

	EntityTypeContainer.String(): EntityTypeContainer,
	EntityTypeLeaf.String():      EntityTypeLeaf,

	if et, ok := entityTypeByName[s]; ok {
		return et, nil
	}
	return 0, fmt.Errorf("unknown entity type %q: must be place, container, or leaf", s)
}
```

```sql
    updated_at         TEXT NOT NULL,
    FOREIGN KEY (parent_id) REFERENCES entities_current(entity_id)
```

```go
// internal/database/entity.go
package database

import (
	"context"
	"database/sql"
	"fmt"
	"time"
)

// Entity represents a single row in entities_current.
type Entity struct {
	EntityID          string
	DisplayName       string
	CanonicalName     string
	EntityType        EntityType
	ParentID          *string
	FullPathDisplay   string
	FullPathCanonical string
	Depth             int
	Status            EntityStatus
	StatusContext     *string
	LastEventID       int64
	UpdatedAt         time.Time
}

// GetEntity returns the entity with the given ID, or an error if not found.
func (d *Database) GetEntity(ctx context.Context, entityID string) (*Entity, error) {
	const query = `
		SELECT entity_id, display_name, canonical_name, entity_type,
		       parent_id, full_path_display, full_path_canonical, depth,
		       status, status_context, last_event_id, updated_at
		FROM entities_current
		WHERE entity_id = ?
	`

	row := d.db.QueryRowContext(ctx, query, entityID)
	return scanEntity(row)
}

// GetEntitiesByCanonicalName returns all entities whose canonical_name matches.
// Returns an empty slice (not an error) when no matches exist.
func (d *Database) GetEntitiesByCanonicalName(ctx context.Context, canonicalName string) ([]*Entity, error) {
	const query = `
		SELECT entity_id, display_name, canonical_name, entity_type,
		       parent_id, full_path_display, full_path_canonical, depth,
		       status, status_context, last_event_id, updated_at
		FROM entities_current
		WHERE canonical_name = ?
		ORDER BY full_path_canonical ASC, entity_id ASC
	`

	rows, err := d.db.QueryContext(ctx, query, canonicalName)
	if err != nil {
		return nil, fmt.Errorf("query entities by canonical name: %w", err)
	}
	defer rows.Close()

	return scanEntities(rows)
}

// GetChildren returns all direct children of the given parent entity.
func (d *Database) GetChildren(ctx context.Context, parentID string) ([]*Entity, error) {
	const query = `
		SELECT entity_id, display_name, canonical_name, entity_type,
		       parent_id, full_path_display, full_path_canonical, depth,
		       status, status_context, last_event_id, updated_at
		FROM entities_current
		WHERE parent_id = ?
		ORDER BY full_path_canonical ASC, entity_id ASC
	`

	rows, err := d.db.QueryContext(ctx, query, parentID)
	if err != nil {
		return nil, fmt.Errorf("query children: %w", err)
	}
	defer rows.Close()

	return scanEntities(rows)
}

// GetDescendants returns all descendants of the given entity (recursive, any depth).
func (d *Database) GetDescendants(ctx context.Context, entityID string) ([]*Entity, error) {
	const query = `
		WITH RECURSIVE descendants AS (
			SELECT entity_id, display_name, canonical_name, entity_type,
			       parent_id, full_path_display, full_path_canonical, depth,
			       status, status_context, last_event_id, updated_at
			FROM entities_current
			WHERE parent_id = ?
			UNION ALL
			SELECT e.entity_id, e.display_name, e.canonical_name, e.entity_type,
			       e.parent_id, e.full_path_display, e.full_path_canonical, e.depth,
			       e.status, e.status_context, e.last_event_id, e.updated_at
			FROM entities_current e
			JOIN descendants d ON e.parent_id = d.entity_id
		)
		SELECT * FROM descendants
		ORDER BY full_path_canonical ASC, entity_id ASC
	`

	rows, err := d.db.QueryContext(ctx, query, entityID)
	if err != nil {
		return nil, fmt.Errorf("query descendants: %w", err)
	}
	defer rows.Close()

	return scanEntities(rows)
}

// ListEntities returns all entities, optionally filtered. Pass empty strings to skip a filter.
func (d *Database) ListEntities(ctx context.Context, underID string, entityType string, status string) ([]*Entity, error) {
	q := `
		SELECT entity_id, display_name, canonical_name, entity_type,
		       parent_id, full_path_display, full_path_canonical, depth,
		       status, status_context, last_event_id, updated_at
		FROM entities_current
		WHERE 1=1
	`
	args := []any{}

	if underID != "" {
		q += " AND (entity_id = ? OR full_path_canonical LIKE (SELECT full_path_canonical || '::%' FROM entities_current WHERE entity_id = ?))"
		args = append(args, underID, underID)
	}
	if entityType != "" {
		q += " AND entity_type = ?"
		args = append(args, entityType)
	}
	if status != "" {
		q += " AND status = ?"
		args = append(args, status)
	}

	q += " ORDER BY full_path_canonical ASC, entity_id ASC"

	rows, err := d.db.QueryContext(ctx, q, args...)
	if err != nil {
		return nil, fmt.Errorf("list entities: %w", err)
	}
	defer rows.Close()

	return scanEntities(rows)
}

// computeEntityPathTx computes full_path_display, full_path_canonical, and depth
// for an entity with the given names and parent, within a transaction.
func (d *Database) computeEntityPathTx(ctx context.Context, tx *sql.Tx, displayName, canonicalName string, parentID *string) (string, string, int, error) {
	if parentID == nil {
		return displayName, canonicalName, 0, nil
	}

	const query = `
		SELECT full_path_display, full_path_canonical, depth
		FROM entities_current
		WHERE entity_id = ?
	`

	var parentDisplay, parentCanonical string
	var parentDepth int

	row := tx.QueryRowContext(ctx, query, *parentID)
	if err := row.Scan(&parentDisplay, &parentCanonical, &parentDepth); err != nil {
		return "", "", 0, fmt.Errorf("parent entity %q not found: %w", *parentID, err)
	}

	return parentDisplay + "::" + displayName,
		parentCanonical + "::" + canonicalName,
		parentDepth + 1,
		nil
}

func scanEntity(row *sql.Row) (*Entity, error) {
	var e Entity
	var entityTypeStr, statusStr string
	var updatedAtStr string

	err := row.Scan(
		&e.EntityID, &e.DisplayName, &e.CanonicalName, &entityTypeStr,
		&e.ParentID, &e.FullPathDisplay, &e.FullPathCanonical, &e.Depth,
		&statusStr, &e.StatusContext, &e.LastEventID, &updatedAtStr,
	)
	if err != nil {
		return nil, fmt.Errorf("scan entity: %w", err)
	}

	e.EntityType, err = ParseEntityType(entityTypeStr)
	if err != nil {
		return nil, err
	}

	e.Status, err = ParseEntityStatus(statusStr)
	if err != nil {
		return nil, err
	}

	e.UpdatedAt, err = time.Parse(time.RFC3339, updatedAtStr)
	if err != nil {
		return nil, fmt.Errorf("parse updated_at: %w", err)
	}

	return &e, nil
}

func scanEntities(rows *sql.Rows) ([]*Entity, error) {
	var entities []*Entity

	for rows.Next() {
		var e Entity
		var entityTypeStr, statusStr, updatedAtStr string
		var err error

		err = rows.Scan(
			&e.EntityID, &e.DisplayName, &e.CanonicalName, &entityTypeStr,
			&e.ParentID, &e.FullPathDisplay, &e.FullPathCanonical, &e.Depth,
			&statusStr, &e.StatusContext, &e.LastEventID, &updatedAtStr,
		)
		if err != nil {
			return nil, fmt.Errorf("scan entity row: %w", err)
		}

		e.EntityType, err = ParseEntityType(entityTypeStr)
		if err != nil {
			return nil, err
		}

		e.Status, err = ParseEntityStatus(statusStr)
		if err != nil {
			return nil, err
		}

		e.UpdatedAt, err = time.Parse(time.RFC3339, updatedAtStr)
		if err != nil {
			return nil, fmt.Errorf("parse updated_at: %w", err)
		}

		entities = append(entities, &e)
	}

	if err := rows.Err(); err != nil {
		return nil, fmt.Errorf("iterate entity rows: %w", err)
	}

	return entities, nil
}
```

```go
func (d *Database) handleEntityCreated(ctx context.Context, tx *sql.Tx, event *Event) error {
	var payload struct {
		EntityID    string  `json:"entity_id"`
		DisplayName string  `json:"display_name"`
		EntityType  string  `json:"entity_type"`
		ParentID    *string `json:"parent_id"`
	}

	if err := json.Unmarshal(event.Payload, &payload); err != nil {
		return fmt.Errorf("unmarshal entity.created payload: %w", err)
	}

	entityType, err := ParseEntityType(payload.EntityType)
	if err != nil {
		return err
	}

	// Validate: place entities may only be nested inside other places.
	if payload.ParentID != nil && entityType == EntityTypePlace {
		if parentErr := validatePlaceParentTx(ctx, tx, *payload.ParentID); parentErr != nil {
			return parentErr
		}
	}

	canonicalName := CanonicalizeString(payload.DisplayName)
	fullPathDisplay, fullPathCanonical, depth, err := d.computeEntityPathTx(
		ctx, tx, payload.DisplayName, canonicalName, payload.ParentID,
	)
	if err != nil {
		return fmt.Errorf("compute path: %w", err)
	}

	const query = `
		INSERT INTO entities_current (
			entity_id, display_name, canonical_name, entity_type, parent_id,
			full_path_display, full_path_canonical, depth,
			status, status_context, last_event_id, updated_at
		) VALUES (?, ?, ?, ?, ?, ?, ?, ?, 'ok', NULL, ?, ?)
	`

	_, err = tx.ExecContext(ctx, query,
		payload.EntityID, payload.DisplayName, canonicalName, payload.EntityType, payload.ParentID,
		fullPathDisplay, fullPathCanonical, depth,
		event.EventID, event.TimestampUTC,
	)
	if err != nil {
		return fmt.Errorf("insert entity: %w", err)
	}

	return nil
}

func (d *Database) handleEntityRenamed(ctx context.Context, tx *sql.Tx, event *Event) error {
	var payload struct {
		EntityID    string `json:"entity_id"`
		DisplayName string `json:"display_name"`
	}

	if err := json.Unmarshal(event.Payload, &payload); err != nil {
		return fmt.Errorf("unmarshal entity.renamed payload: %w", err)
	}

	canonicalName := CanonicalizeString(payload.DisplayName)

	// Get the entity's current parent to recompute paths.
	var parentID *string
	row := tx.QueryRowContext(ctx, `SELECT parent_id FROM entities_current WHERE entity_id = ?`, payload.EntityID)
	if err := row.Scan(&parentID); err != nil {
		return fmt.Errorf("get entity for rename: %w", err)
	}

	fullPathDisplay, fullPathCanonical, depth, err := d.computeEntityPathTx(
		ctx, tx, payload.DisplayName, canonicalName, parentID,
	)
	if err != nil {
		return fmt.Errorf("compute path: %w", err)
	}

	_, err = tx.ExecContext(ctx, `
		UPDATE entities_current
		SET display_name = ?, canonical_name = ?,
		    full_path_display = ?, full_path_canonical = ?, depth = ?,
		    last_event_id = ?, updated_at = ?
		WHERE entity_id = ?
	`, payload.DisplayName, canonicalName,
		fullPathDisplay, fullPathCanonical, depth,
		event.EventID, event.TimestampUTC,
		payload.EntityID,
	)
	if err != nil {
		return fmt.Errorf("update renamed entity: %w", err)
	}

	return d.propagatePathChanges(ctx, tx, payload.EntityID, event)

func (d *Database) handleEntityReparented(ctx context.Context, tx *sql.Tx, event *Event) error {

		EntityID string  `json:"entity_id"`

		return fmt.Errorf("unmarshal entity.reparented payload: %w", err)

	// Validate: place entities may not be reparented under non-places.
	if payload.ParentID != nil {
		var entityTypeStr string

		if err := row.Scan(&entityTypeStr); err != nil {

		entityType, err := ParseEntityType(entityTypeStr)
		if err != nil {
			return err
		}

	}

	fullPathDisplay, fullPathCanonical, depth, err := d.computeEntityPathTx(
		ctx, tx, displayName, canonicalName, payload.ParentID,

func (d *Database) handleEntityRemoved(ctx context.Context, tx *sql.Tx, event *Event) error {
	var payload struct {
		EntityID string `json:"entity_id"`
	}

	if err := json.Unmarshal(event.Payload, &payload); err != nil {
		return fmt.Errorf("unmarshal entity.removed payload: %w", err)
	}

	// Reject if any non-removed children exist.
	var childCount int
	row := tx.QueryRowContext(ctx, `
		SELECT COUNT(*) FROM entities_current
		WHERE parent_id = ? AND status != 'removed'
	`, payload.EntityID)
	if err := row.Scan(&childCount); err != nil {
		return fmt.Errorf("check children: %w", err)
	}
	if childCount > 0 {
		return fmt.Errorf("entity %q has %d non-removed children; remove them first", payload.EntityID, childCount)
	}

	_, err := tx.ExecContext(ctx, `
		UPDATE entities_current
		SET status = 'removed', status_context = NULL,
		    last_event_id = ?, updated_at = ?
		WHERE entity_id = ?
	`, event.EventID, event.TimestampUTC, payload.EntityID)
	if err != nil {
		return fmt.Errorf("mark entity removed: %w", err)
	}

	return nil
}

// propagatePathChanges recomputes and updates full_path_* for all descendants
// of entityID, and appends entity.path_changed events for each.
func (d *Database) propagatePathChanges(ctx context.Context, tx *sql.Tx, entityID string, triggerEvent *Event) error {
	descendants, err := d.getDescendantsTx(ctx, tx, entityID)
	if err != nil {
		return fmt.Errorf("get descendants: %w", err)
	}

	for _, desc := range descendants {
		parentDisplay, parentCanonical, parentDepth, pathErr := d.getParentPathTx(ctx, tx, desc.ParentID)
		if pathErr != nil {
			return pathErr
		}

		newPathDisplay := parentDisplay + "::" + desc.DisplayName
		newPathCanonical := parentCanonical + "::" + desc.CanonicalName
		newDepth := parentDepth + 1

		_, err = tx.ExecContext(ctx, `
			UPDATE entities_current
			SET full_path_display = ?, full_path_canonical = ?, depth = ?,
			    last_event_id = ?, updated_at = ?
			WHERE entity_id = ?
		`, newPathDisplay, newPathCanonical, newDepth,
			triggerEvent.EventID, triggerEvent.TimestampUTC,
			desc.EntityID,
		)
		if err != nil {
			return fmt.Errorf("update descendant path: %w", err)
		}

		// Append a derived entity.path_changed event to the event log.
		pathPayload := map[string]any{
			"entity_id":           desc.EntityID,
			"full_path_display":   newPathDisplay,
			"full_path_canonical": newPathCanonical,
			"depth":               newDepth,
		}
		if appendErr := d.appendDerivedEventTx(ctx, tx, EntityPathChangedEvent, triggerEvent.ActorUserID, pathPayload, triggerEvent.Note); appendErr != nil {
			return fmt.Errorf("append path_changed event: %w", appendErr)
		}
	}

	return nil
}

func (d *Database) getDescendantsTx(ctx context.Context, tx *sql.Tx, entityID string) ([]*Entity, error) {
	const query = `
		WITH RECURSIVE descendants AS (
			SELECT entity_id, display_name, canonical_name, entity_type,
			       parent_id, full_path_display, full_path_canonical, depth,
			       status, status_context, last_event_id, updated_at
			FROM entities_current WHERE parent_id = ?
			UNION ALL
			SELECT e.entity_id, e.display_name, e.canonical_name, e.entity_type,
			       e.parent_id, e.full_path_display, e.full_path_canonical, e.depth,
			       e.status, e.status_context, e.last_event_id, e.updated_at
			FROM entities_current e
			JOIN descendants d ON e.parent_id = d.entity_id
		)
		SELECT * FROM descendants ORDER BY depth ASC, full_path_canonical ASC
	`

	rows, err := tx.QueryContext(ctx, query, entityID)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	return scanEntities(rows)
}

func (d *Database) getParentPathTx(ctx context.Context, tx *sql.Tx, parentID *string) (string, string, int, error) {
	if parentID == nil {
		return "", "", -1, nil
	}

	var display, canonical string
	var depth int
	row := tx.QueryRowContext(ctx, `SELECT full_path_display, full_path_canonical, depth FROM entities_current WHERE entity_id = ?`, *parentID)
	if err := row.Scan(&display, &canonical, &depth); err != nil {
		return "", "", 0, fmt.Errorf("get parent path: %w", err)
	}

	return display, canonical, depth, nil
}

func validatePlaceParentTx(ctx context.Context, tx *sql.Tx, parentID string) error {
	var parentTypeStr string
	row := tx.QueryRowContext(ctx, `SELECT entity_type FROM entities_current WHERE entity_id = ?`, parentID)
	if err := row.Scan(&parentTypeStr); err != nil {
		return fmt.Errorf("parent entity %q not found: %w", parentID, err)
	}

	parentType, err := ParseEntityType(parentTypeStr)
	if err != nil {
		return err
	}

	if parentType != EntityTypePlace {
		return fmt.Errorf("a place entity can only be nested inside another place, not %q", parentTypeStr)
	}

	return nil
}

// appendDerivedEventTx writes a derived event directly within an existing transaction.
// Used for entity.path_changed events that are generated as side effects.
func (d *Database) appendDerivedEventTx(ctx context.Context, tx *sql.Tx, eventType EventType, actorUserID string, payload any, note string) error {
	payloadJSON, err := json.Marshal(payload)
	if err != nil {
		return fmt.Errorf("marshal derived event payload: %w", err)
	}

	_, err = tx.ExecContext(ctx, `
		INSERT INTO events (event_type, actor_user_id, payload, note, timestamp_utc)
		VALUES (?, ?, ?, ?, ?)
	`, eventType.String(), actorUserID, payloadJSON, note, time.Now().UTC().Format(time.RFC3339))
	if err != nil {
		return fmt.Errorf("insert derived event: %w", err)
	}

	return nil
}
```

```go
// NewAddCmd returns the add command with the given DB (used in tests).
func NewAddCmd(db addDB) *cobra.Command {
	cmd := &cobra.Command{
		Use:   "add <name>",
		Short: "Add an entity to the inventory",
		Long: `Add a new entity. By default, entities are containers (movable, can hold things).
Use --type place for immovable locations like rooms or shelves.`,
		Args: cobra.ExactArgs(1),
		RunE: func(cmd *cobra.Command, args []string) error {
			return runAdd(cmd, args, db)
		},
	}

	cmd.Flags().StringP("in", "i", "", "Parent entity ID or unambiguous name")
	cmd.Flags().StringP("type", "t", "container", "Entity type: place, container, or leaf")

	return cmd
}

type addResult struct {
	EntityID string `json:"entity_id"`
	Path     string `json:"path"`
}

func runAdd(cmd *cobra.Command, args []string, db addDB) error {
	ctx := cmd.Context()
	name := args[0]

	inFlag, _ := cmd.Flags().GetString("in")
	typeFlag, _ := cmd.Flags().GetString("type")

	entityType, err := database.ParseEntityType(typeFlag)
	if err != nil {
		return err
	}

	var parentID *string
	if inFlag != "" {
		resolved, resolveErr := resolveParent(ctx, db, inFlag)
		if resolveErr != nil {
			return resolveErr
		}
		parentID = &resolved
	}

	entityID := nanoid.MustNew()
	actorUserID := cli.GetActorUserID(ctx)

	payload := map[string]any{
		"entity_id":    entityID,
		"display_name": name,
		"entity_type":  entityType.String(),
		"parent_id":    parentID,
	}

	if _, err = db.AppendEvent(ctx, database.EntityCreatedEvent, actorUserID, payload, ""); err != nil {
		return fmt.Errorf("failed to create entity: %w", err)
	}

	entity, err := db.GetEntity(ctx, entityID)
	if err != nil {
		return fmt.Errorf("failed to retrieve created entity: %w", err)
	}

	cfg := cli.MustGetConfig(ctx)
	out := cli.NewOutputWriterFromConfig(cmd.OutOrStdout(), cmd.ErrOrStderr(), cfg)

	if cfg.IsJSON() {
		return out.JSON(addResult{EntityID: entityID, Path: entity.FullPathDisplay})
	}

	out.Success(fmt.Sprintf("Added %q (%s) at path %s", name, entityType, entity.FullPathDisplay))
	out.KeyValue("ID", entityID)

	return nil
}

// resolveParent resolves --in flag value to an entity ID.
// Accepts a nanoid directly, or a canonical name if unambiguous.
func resolveParent(ctx context.Context, db addDB, input string) (string, error) {
	// Try direct ID lookup first.
	if e, err := db.GetEntity(ctx, input); err == nil {
		return e.EntityID, nil
	}

	// Fall back to name lookup.
	canonical := database.CanonicalizeString(input)
	matches, err := db.GetEntitiesByCanonicalName(ctx, canonical)
	if err != nil {
		return "", fmt.Errorf("resolve parent %q: %w", input, err)
	}

	switch len(matches) {
	case 0:
		return "", fmt.Errorf("no entity found with name or ID %q", input)
	case 1:
		return matches[0].EntityID, nil
	default:
		return "", fmt.Errorf("ambiguous parent %q: %d entities match; use an entity ID instead", input, len(matches))
	}
}
```

```go
func NewMoveCmd(db moveDB) *cobra.Command {
	cmd := &cobra.Command{
		Use:   "move <entity-id>",
		Short: "Move an entity to a new parent",
		Args:  cobra.ExactArgs(1),
		RunE: func(cmd *cobra.Command, args []string) error {
			return runMove(cmd, args, db)
		},
	}

	cmd.Flags().StringP("to", "t", "", "Destination entity ID (REQUIRED)")
	_ = cmd.MarkFlagRequired("to")

	return cmd
}

type moveResult struct {
	EntityID    string `json:"entity_id"`
	DisplayName string `json:"display_name"`
	OldPath     string `json:"old_path"`
	NewPath     string `json:"new_path"`
}

func runMove(cmd *cobra.Command, args []string, db moveDB) error {
	ctx := cmd.Context()
	entityID := args[0]
	toID, _ := cmd.Flags().GetString("to")

	entity, err := db.GetEntity(ctx, entityID)
	if err != nil {
		return fmt.Errorf("entity %q not found: %w", entityID, err)
	}

	if entity.EntityType == database.EntityTypePlace {
		return fmt.Errorf("place entities cannot be moved: %q is a place", entity.DisplayName)
	}

	dest, err := db.GetEntity(ctx, toID)
	if err != nil {
		return fmt.Errorf("destination %q not found: %w", toID, err)
	}

	oldPath := entity.FullPathDisplay

	payload := map[string]any{
		"entity_id": entityID,
		"parent_id": dest.EntityID,
	}

	actorUserID := cli.GetActorUserID(ctx)
	if _, err = db.AppendEvent(ctx, database.EntityReparentedEvent, actorUserID, payload, ""); err != nil {
		return fmt.Errorf("failed to move entity: %w", err)
	}

	// Re-fetch to get updated path.
	updated, err := db.GetEntity(ctx, entityID)
	if err != nil {
		return fmt.Errorf("failed to retrieve updated entity: %w", err)
	}

	cfg := cli.MustGetConfig(ctx)
	out := cli.NewOutputWriterFromConfig(cmd.OutOrStdout(), cmd.ErrOrStderr(), cfg)

	if cfg.IsJSON() {
		return out.JSON(moveResult{
			EntityID:    entityID,
			DisplayName: entity.DisplayName,
			OldPath:     oldPath,
			NewPath:     updated.FullPathDisplay,
		})
	}

	out.Success(fmt.Sprintf("Moved %q: %s → %s", entity.DisplayName, oldPath, updated.FullPathDisplay))
	return nil
}
```

```go
func NewStatusCmd(db statusDB) *cobra.Command {
	cmd := &cobra.Command{

		Short: "Change the status of an entity",
		Long: `Change the status of an entity.

Valid statuses: ok, borrowed, missing, loaned, removed

Examples:
  wherehouse status <id> --set loaned --note "loaned to Alice"
  wherehouse status <id> --set ok`,
		Args: cobra.ExactArgs(1),
		RunE: func(cmd *cobra.Command, args []string) error {
			return runStatus(cmd, args, db)
		},
	}

	cmd.Flags().StringP("set", "s", "", "New status: ok, borrowed, missing, loaned, removed (REQUIRED)")
	_ = cmd.MarkFlagRequired("set")
	cmd.Flags().StringP("note", "n", "", "Optional context note (e.g. who borrowed it)")

	return cmd
}

type statusResult struct {
	EntityID      string  `json:"entity_id"`
	DisplayName   string  `json:"display_name"`
	Status        string  `json:"status"`
	StatusContext *string `json:"status_context,omitempty"`
}

func runStatus(cmd *cobra.Command, args []string, db statusDB) error {
	ctx := cmd.Context()
	entityID := args[0]
	setFlag, _ := cmd.Flags().GetString("set")
	noteFlag, _ := cmd.Flags().GetString("note")

	newStatus, err := database.ParseEntityStatus(setFlag)
	if err != nil {
		return err
	}

	entity, err := db.GetEntity(ctx, entityID)
	if err != nil {
		return fmt.Errorf("entity %q not found: %w", entityID, err)
	}

	var statusContext *string
	if noteFlag != "" {
		statusContext = &noteFlag
	}

	payload := map[string]any{
		"entity_id":      entityID,
		"status":         newStatus.String(),
		"status_context": statusContext,
	}

	actorUserID := cli.GetActorUserID(ctx)
	if _, err = db.AppendEvent(ctx, database.EntityStatusChangedEvent, actorUserID, payload, ""); err != nil {
		return fmt.Errorf("failed to update status: %w", err)
	}

	cfg := cli.MustGetConfig(ctx)
	out := cli.NewOutputWriterFromConfig(cmd.OutOrStdout(), cmd.ErrOrStderr(), cfg)

	if cfg.IsJSON() {
		return out.JSON(statusResult{
			EntityID:      entityID,
			DisplayName:   entity.DisplayName,
			Status:        newStatus.String(),
			StatusContext: statusContext,
		})
	}

	msg := fmt.Sprintf("Status of %q set to %s", entity.DisplayName, newStatus)
	if noteFlag != "" {
		msg += fmt.Sprintf(" (%s)", noteFlag)
	}
	out.Success(msg)

	return nil
}
```

```go
// cmd/rename/rename.go
package rename

import (
	"fmt"

	"github.com/asphaltbuffet/wherehouse/internal/cli"
	"github.com/asphaltbuffet/wherehouse/internal/database"
	"github.com/spf13/cobra"
)

func NewDefaultRenameCmd() *cobra.Command {
	db, err := database.OpenDefault()
	if err != nil {
		cmd := &cobra.Command{Use: "rename"}
		cmd.RunE = func(cmd *cobra.Command, args []string) error {
			return fmt.Errorf("failed to open database: %w", err)
		}
		return cmd
	}
	return NewRenameCmd(db)
}

func NewRenameCmd(db renameDB) *cobra.Command {
	cmd := &cobra.Command{
		Use:   "rename <entity-id>",
		Short: "Rename an entity",
		Args:  cobra.ExactArgs(1),
		RunE: func(cmd *cobra.Command, args []string) error {
			return runRename(cmd, args, db)
		},
	}

	cmd.Flags().StringP("to", "t", "", "New display name (REQUIRED)")
	_ = cmd.MarkFlagRequired("to")

	return cmd
}

type renameResult struct {
	EntityID    string `json:"entity_id"`
	OldName     string `json:"old_name"`
	NewName     string `json:"new_name"`
	NewPath     string `json:"new_path"`
}

func runRename(cmd *cobra.Command, args []string, db renameDB) error {
	ctx := cmd.Context()
	entityID := args[0]
	toFlag, _ := cmd.Flags().GetString("to")

	entity, err := db.GetEntity(ctx, entityID)
	if err != nil {
		return fmt.Errorf("entity %q not found: %w", entityID, err)
	}

	oldName := entity.DisplayName

	payload := map[string]any{
		"entity_id":    entityID,
		"display_name": toFlag,
	}

	actorUserID := cli.GetActorUserID(ctx)
	if _, err = db.AppendEvent(ctx, database.EntityRenamedEvent, actorUserID, payload, ""); err != nil {
		return fmt.Errorf("failed to rename entity: %w", err)
	}

	updated, err := db.GetEntity(ctx, entityID)
	if err != nil {
		return fmt.Errorf("failed to retrieve renamed entity: %w", err)
	}

	cfg := cli.MustGetConfig(ctx)
	out := cli.NewOutputWriterFromConfig(cmd.OutOrStdout(), cmd.ErrOrStderr(), cfg)

	if cfg.IsJSON() {
		return out.JSON(renameResult{
			EntityID: entityID,
			OldName:  oldName,
			NewName:  toFlag,
			NewPath:  updated.FullPathDisplay,
		})
	}

	out.Success(fmt.Sprintf("Renamed %q to %q (path: %s)", oldName, toFlag, updated.FullPathDisplay))
	return nil
}
```

```go
package rename_test

import (
	"bytes"
	"context"
	"encoding/json"
	"testing"

	"github.com/asphaltbuffet/wherehouse/cmd/rename"
	"github.com/asphaltbuffet/wherehouse/internal/database"
	"github.com/asphaltbuffet/wherehouse/internal/nanoid"
	"github.com/stretchr/testify/assert"
	"github.com/stretchr/testify/require"
)

func openTestDB(t *testing.T) (*database.Database, context.Context) {
	t.Helper()
	ctx := context.Background()
	db, err := database.Open(database.Config{
		Path:        ":memory:",
		BusyTimeout: database.DefaultBusyTimeout,
		AutoMigrate: true,
	})
	require.NoError(t, err)
	t.Cleanup(func() { db.Close() })
	return db, ctx
}

func appendEntity(t *testing.T, db *database.Database, ctx context.Context, id, name, entityType string, parentID *string) {
	t.Helper()
	payload, _ := json.Marshal(map[string]any{
		"entity_id":    id,
		"display_name": name,
		"entity_type":  entityType,
		"parent_id":    parentID,
	})
	_, err := db.AppendEvent(ctx, database.EntityCreatedEvent, "testuser", json.RawMessage(payload), "")
	require.NoError(t, err)
}

func TestRenameEntity_UpdatesPathAndDescendants(t *testing.T) {
	db, ctx := openTestDB(t)

	garageID := nanoid.MustNew()
	toolboxID := nanoid.MustNew()
	screwdriverID := nanoid.MustNew()

	appendEntity(t, db, ctx, garageID, "Garage", "place", nil)
	appendEntity(t, db, ctx, toolboxID, "Toolbox", "container", &garageID)
	appendEntity(t, db, ctx, screwdriverID, "screwdriver", "container", &toolboxID)

	cmd := rename.NewRenameCmd(db)
	cmd.SetContext(ctx)
	var out bytes.Buffer
	cmd.SetOut(&out)
	cmd.SetArgs([]string{toolboxID, "--to", "Big Toolbox"})

	require.NoError(t, cmd.Execute())
	assert.Contains(t, out.String(), "Big Toolbox")

	toolbox, err := db.GetEntity(ctx, toolboxID)
	require.NoError(t, err)
	assert.Equal(t, "Garage::Big Toolbox", toolbox.FullPathDisplay)

	sd, err := db.GetEntity(ctx, screwdriverID)
	require.NoError(t, err)
	assert.Equal(t, "Garage::Big Toolbox::screwdriver", sd.FullPathDisplay)
}

func TestRenameEntity_NotFound_Errors(t *testing.T) {
	db, ctx := openTestDB(t)
	cmd := rename.NewRenameCmd(db)
	cmd.SetContext(ctx)
	cmd.SetArgs([]string{"doesnotexist", "--to", "NewName"})
	err := cmd.Execute()
	require.Error(t, err)
}
```
